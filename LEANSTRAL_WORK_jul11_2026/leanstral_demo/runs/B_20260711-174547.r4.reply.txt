| nil => rfl
| cons b bs ih =>
  simp [revcomp, comp]
  have h_sym_comp : Function.Involutive symComp := by decide
  have h_comp_invol (b : Base) : comp (comp b) = b := by
    rcases b with ⟨sym, masked⟩; simp [comp, h_sym_comp sym]
  rw [h_comp_invol b, ih]
