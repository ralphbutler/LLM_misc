  have hdec : (a.lo < b.hi ∧ b.lo < a.hi) := by

  set x := max a.lo b.lo with hxdef
  have hax : a.lo ≤ x := by
    simp [hxdef]
  have hbx : b.lo ≤ x := by
    simp [hxdef]

    have : max a.lo b.lo < a.hi := max_lt ha hlt2
    simpa [hxdef] using this
  have hxb : x < b.hi := by
    have : max a.lo b.lo < b.hi := max_lt hlt1 hb
    simpa [hxdef] using this

  simp [overlap, hlt1, hlt2]
